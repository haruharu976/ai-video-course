STEP 02

企画エージェントを作る

ClaudeのAPIを使ってトレンド分析と台本生成を自動化する。

PYTHON — PLANNING AGENT 
import anthropic
import os

client = anthropic.Anthropic(api_key=os.environ["ANTHROPIC_API_KEY"])

def create_script(topic: str) -> str:
  message = client.messages.create(
    model="claude-sonnet-4-6",
    max_tokens=1024,
    messages=[{
      "role": "user",
      "content": f"""
        テーマ「{topic}」について
        30秒のショート動画の台本を作成してください。
        フック（0-3秒）、本編（4-25秒）、CTA（26-30秒）
        の構成でJSON形式で出力。
      """
    }]
  )
  return message.content[0].text

script = create_script("AI動画制作の革命")
print(script)
